\section{Singapore Government sources}

\subsection{Primary normative sources}

\begin{enumerate}
  \item IMDA. \textit{Model AI Governance Framework for Agentic AI}.
        Singapore: Infocomm Media Development Authority, January 2026.
        
        \begin{itemize}
          \item Pillar 1: Assess and Bound Risks Upfront
          \item Pillar 2: Make Humans Meaningfully Accountable
          \item Pillar 3: Implement Technical Controls and Processes
          \item Pillar 4: Enable End-User Responsibility
        \end{itemize}
  
  \item IMDA. \textit{Starter Kit for Testing LLM-Based Applications for Safety
        and Reliability}. Singapore: Infocomm Media Development Authority,
        January 2026.
  
  \item PDPC. \textit{Model AI Governance Framework, Second Edition}.
        Singapore: Personal Data Protection Commission, January 2020.
  
  \item PDPC. \textit{Advisory Guidelines on Use of Personal Data in AI
        Recommendation and Decision Systems}. Singapore: Personal Data
        Protection Commission, April 2024.
\end{enumerate}
