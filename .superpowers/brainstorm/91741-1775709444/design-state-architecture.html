AppStore Extension

app.store.ts — new mode slice (additions only)
type AppMode = 'chat' | 'cowork' | 'training';

// Added to signalStore()
activeMode: 'chat', // initial state

// New computed properties
withComputed(store => ({
  aiCapabilities: computed(() => getModeCapabilities(store.activeMode())),
  routeRelevance: computed(() => getRouteRelevance(store.activeMode())),
  contextPills:  computed(() => getContextPills(store.activeMode())),
  modeThemeClass: computed(() => `mode-${store.activeMode()}`),
}))

// New method
withMethods(store => ({
  setMode(mode: AppMode) { patchState(store, { activeMode: mode }); }
}))
